var config = new Configuration
{
	ClientId = clientId,
	ClientSecret = clientSecret,
	ApiBaseUrl = baseUrl,
};

var wordsApi = new WordsApi(config);
using (var inputStream = new FileStream(pathToDocFile, FileMode.Open))
{
	var uploadFileRequest = new UploadFileRequest(inputStream, "template.{{ lower .Page.Params.informat }}");
	wordsApi.UploadFile(uploadFileRequest);
}

var saveOptionsData = new SaveOptionsData { SaveFormat = "{{ lower .Page.Params.outformat }}", FileName = "output.{{ lower .Page.Params.outformat }}" };
var request = new SaveAsRequest("output.{{ lower .Page.Params.outformat }}", saveOptionsData);
wordsApi.SaveAs(request);
